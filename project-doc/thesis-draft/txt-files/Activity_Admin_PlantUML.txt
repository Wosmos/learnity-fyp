@startuml
title Activity Diagram: Admin Workflow (System Governance)

start
:Admin Dashboard;

fork
    partition "User Management" {
        :Review Teacher Requests;
        :Check Documents;
        if (Valid?) then (yes)
            :Approve Teacher;
            :Send Notification;
        else (no)
            :Reject Teacher;
            :Send Rejection Email;
        endif
    }
fork again
    partition "Financial Management" {
        :View Payment Queue;
        :Validate Screenshots;
        if (Payment Verified?) then (yes)
            :Mark as Verified;
            :Unlock Student Course;
        else (no)
            :Reject Payment;
        endif
    }
end fork

:Return to Dashboard;
stop
@enduml
