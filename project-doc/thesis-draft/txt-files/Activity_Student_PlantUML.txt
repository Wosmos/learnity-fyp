@startuml
title Activity Diagram: Student Workflow (Enrollment to Learning)

start
:Login;
:Browse Catalog;
:Filter Subjects (Grade/Subject);
:Select Course;

partition "Enrollment Process" {
    if (Is Course Free?) then (no)
        :Payment Required;
        :Upload Screenshot (Manual Payment);
        :Wait for Admin Verification;
        if (Admin Approves?) then (yes)
            :Access Granted;
        else (no)
            :Notification: Payment Rejected;
            stop
        endif
    else (yes)
        :Enroll Immediately;
    endif
}

partition "Learning Loop" {
    :Join Classroom;
    fork
        :Watch Video Lesson;
    fork again
        :Join Real-Time Chat;
    end fork
    
    :Take Quiz;
    if (Pass Quiz?) then (yes)
        :Earn XP & Streak;
        :Update Leaderboard;
        :Unlock Next Lesson;
    else (no)
        :Review Lesson Material;
    endif
}

stop
@enduml
