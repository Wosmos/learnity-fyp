@startuml
title Activity Diagram: Teacher Workflow (Verification & Creation)

start
:Register as User;
:Request Teacher Access;

partition "Verification Process" {
    :Upload Degree/Documents;
    :Status: Pending Admin Review;
    if (Documents Valid?) then (yes)
        :Approved (Teacher Role Granted);
    else (no)
        :Rejected;
        stop
    endif
}

partition "Course Creation" {
    :Create Course Draft;
    repeat
        :Add Section;
        :Add Lesson (Video/Text);
    repeat while (More Content?)
    :Publish Course;
}

partition "Live Notification" {
    :Manage Live Sessions;
    :Start Video Stream (100ms);
    fork
        :Broadcast Video;
    fork again
        :Moderate Chat (GetStream);
    end fork
    :End Session;
}

stop
@enduml
